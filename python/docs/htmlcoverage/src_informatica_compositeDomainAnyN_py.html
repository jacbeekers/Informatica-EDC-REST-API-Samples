193def assemble_combinations(domain_combinations: list, domain_objects: dict): 

194 all_combinations = list() 

195 for domain_combination in domain_combinations: 

196 combi_list = list() 

197 combi_entry = dict() 

198 for a_domain in sorted(domain_combination): 

199 # domain_object = domain_objects[a_domain] 

200 combi_list.append(domain_objects[a_domain]) 

201 combi_entry["compositeDomainRuleElements"] = combi_list 

202 combi_entry["ruleOperation"] = "AND" 

203 combi_entry["valid"] = "true" 

204 # separator = " AND " 

205 # test = separator.join(elems for elems in domain_combinations) 

206 combi_entry["inferenceCondition"] = " AND ".join(sorted(domain_combination)) 

207 all_combinations.append(combi_entry) 

208 

209 return all_combinations

212def getDomainComponents(domain_def: str) -> dict: 

213 unique_objects = dict() 

214 obj_count = 0

216 for obj_ref in inf_set["compositeDomainRuleElements"]: 

217 obj_count += 1 

218 obj_name = obj_ref["cDomainElementName"] 

219 unique_objects[obj_name] = obj_ref 

220 

221 print( 

222 f"unique objects found {len(unique_objects)} out of {obj_count} total objects" 

223 ) 

224 print(f"keys={sorted(unique_objects.keys())}")
